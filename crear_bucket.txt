import boto3

def lambda_handler(event, context):
    # Entrada
    s3 = boto3.client('s3')

    # Proceso
    s3.create_bucket(
        Bucket=event['name'], 
        CreateBucketConfiguration={
            'LocationConstraint': 'us-west-1'
        }
    )
    
    # Salida
    return {
        'statusCode': 200,
        'body': "Se cre√≥ correctamente el bucket"
    }
